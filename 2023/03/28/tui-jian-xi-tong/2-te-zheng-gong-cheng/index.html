import pandas as pd
import matplotlib.pyplot as plt

# 读取csv
BASE_PATH = '../recall_service_tmp/archive'
anime_df = pd.read_csv(f'{BASE_PATH}/anime.csv', index_col='anime_id')
rating_df = pd.read_csv(f'{BASE_PATH}/rating.csv')
# 数据格式
print(anime_df.dtypes)
print(rating_df.dtypes)
# 数据规模
uniq_anime_count = rating_df['anime_id'].nunique()
uniq_user_count = rating_df['user_id'].nunique()
print(uniq_anime_count)
print(uniq_user_count)
## 总评价数
print(rating_df.shape[0])
# 数据分布
uniq_ratings = rating_df['rating'].nunique()
print(uniq_ratings)  ## 11(-1是未评价,1~10是具体分数)
## 直方图
rating_df['rating'].plot.hist(bins=2 * uniq_ratings + 1)
## 展示统计图
## 可以看到,高评分比较多(符合假设: 遇到喜欢的才喜欢评价)
plt.show()
## describe可以得出统计数据 count mean(平均分) std(标准差) min 25% 50%(中位数) ...
print(rating_df.query('rating > 0')['rating'].describe())
valid_rating_df = rating_df[rating_df['rating'] > 0]
# 每部动漫的评分数
rating_groupby = valid_rating_df.groupby(['anime_id']).count()
rating_groupby.plot.hist(bins=100, grid=True, figsize=[12, 8], alpha=0.5)
## 评分多的动漫少(x轴:评价数,y轴anime数量)
plt.show()
print(rating_groupby.describe())
# 每个用户的评分
## 根据userid分组,计数
user_rating_count = valid_rating_df.groupby(['user_id']).count()
user_rating_count.plot.hist(bins=100, grid=True, figsize=[12, 8], alpha=0.5)
plt.show()
print(user_rating_count.describe())
# 动漫类型分布
genres = {}
## 遍历
for (index, row) in anime_df.iterrows():
    try:
        ## 获取每一行的genre列的数据(是一个数组),用 , 分开
        anime_genres = [g.strip() for g in row['genre'].split(',')]
        for genre in anime_genres:
            if genre not in genres:
                ## 初始化
                genres[genre] = 0
            genres[genre] += 1
    except:
        pass
print(genres)
## 创建图像
fig = plt.figure(figsize=(20, 10))
## 轴
ax = fig.add_axes([0, 0, 1, 1])
## 排序 -> 类型: 该类型影片数量
tuples = sorted(genres.items(), key=lambda x: x[1])
## 图中标签,即为类型,是tuples的[0]
labels = [t[0] for t in tuples]
values = [t[1] for t in tuples]
ax.bar(labels, values)
## 标签旋转,字体放大
plt.xticks(rotation=70, fontsize=15)
plt.show()

# 评分前几名的动漫(自己写的)
anime_ratings = {}
## 遍历
for (index, row) in rating_df.iterrows():
    try:
        anime_rating = row['rating']
        anime_id = row['anime_id']
        if anime_id not in anime_ratings:
            anime_ratings[anime_id] = 0
        ## 记录最高评分
        if anime_rating > anime_ratings[anime_id]:
            anime_ratings[anime_id] = anime_rating
    except:
        pass
print(anime_ratings)

# 评分数量和评分高低的关系
groupby_df = valid_rating_df \
    .groupby(['anime_id']) \
    .agg({
    ## 取平均分和评分count
    'rating': ['count', 'mean']
}) \
    .droplevel(axis=1, level=0)  ## 去掉一个轴
print(groupby_df)
## 散点图, x: 评分数,y: 平均分
groupby_df.plot.scatter(x='count', y='mean', grid=True, figsize=[12, 8])
plt.show()

import pyspark
from pyspark import SparkContext
from pyspark.sql import SparkSession
from pyspark.sql.functions import *
from pyspark.ml.feature import StringIndexer, MinMaxScaler, VectorAssembler
from pyspark.ml import Pipeline
import numpy as np

# 用spark进行特征工程
spark = SparkSession \
    .builder \
    .appName('feat-eng') \
    .getOrCreate()
BASE_PATH = '../../recall_service_tmp/archive'
anime_df = spark.read.csv(f'{BASE_PATH}/anime.csv', header=True, inferSchema=True)
anime_df.printSchema()
# Multi-hot encode genres
## 给不同的动漫类型一个编码
## 1 withColumn定义一个新列, 根据 , 分割genre列的数据(一个字符串),转成数组, explode平展 [动漫id1|类型1] [动漫id1|类型2] [动漫id2|类型1]
genre_df = anime_df \
    .withColumn('genre_item', explode(split(col('genre'), '[,]'))) \
    .withColumn('genre_item', trim(col('genre_item')))  ## 去掉首尾空格
genre_df.select(['anime_id', 'genre_item']).show(10)
# 编码 -> 动漫类型: 对应编码
## inputCol 编码列, outputCol 结果输出列
string_indexer = StringIndexer(inputCol='genre_item', outputCol='genre_index')
## fit 训练
## transform 将fit的结果作用在xxx上
genre_indexed_df = string_indexer \
    .fit(genre_df) \
    .transform(genre_df) \
    .withColumn('genre_index', col('genre_index').cast('int'))  ## 转为int类型
genre_indexed_df[['anime_id', 'genre_item', 'genre_index']].show(10)
# 将同一个动漫id对应的动漫类型聚合(原本的genre是动漫类型字符串,现在变成用动漫类型编码)
## multihot编码就是在multihot里的下标
pre_multihot_df = genre_indexed_df \
    .groupby('anime_id') \
    .agg(collect_list('genre_index').alias('genre_indexes'))  ## 聚合同一个分组的genre_index为list
pre_multihot_df.show(10)
max_genre_index = genre_indexed_df \
    .agg(max(col('genre_index'))).head()['max(genre_index)']  ## 动漫类型编码最大的index,从而可以得知有多少种类型


# 映射动漫类型为矩阵(动漫类型index对应下标的值为1,其他位置为0)
@udf(returnType='array<int>')  ## 优化,分发函数给worker节点,才能被spark执行
def multihot_list(l, max_index):
    ## 定义0向量(矩阵),长度为max_index+1(动漫类型数)
    fill = np.zeros(max_index + 1, dtype=np.int32)
    ## l是动漫类型列 [1,4,5,...]
    for i in l:
        fill[i] = 1
    return fill.tolist()


multihot_df = pre_multihot_df \
    .withColumn('genre_multihot', multihot_list(col('genre_indexes'), lit(max_genre_index)))
multihot_df.printSchema()
## multihot_df.show(10)
# Average Rating Min-Max Scale
# 获取数据
rating_df = spark.read.csv(f'{BASE_PATH}/rating.csv', header=True, inferSchema=True) \
    .filter(col('rating') > 0)
ave_rating_df = rating_df \
    .groupby('anime_id') \
    .agg(mean('rating').alias('ave_rating'))  ## 评价得分
ave_rating_df.show(10)
## MIN-MAX 需要输入的类型是向量
## 转向量
vec_assembler = VectorAssembler(inputCols=['ave_rating'], outputCol='ave_rating_vec')
### 归一化
ave_rating_scaler = MinMaxScaler(inputCol='ave_rating_vec', outputCol='ave_rating_scaled')
## vec_assembler和ave_rating_scaler都要fit和transfer,所以打包到一起,统一处理
pipeline = Pipeline(stages=[vec_assembler, ave_rating_scaler])
rating_scaled_df = pipeline \
    .fit(ave_rating_df) \
    .transform(ave_rating_df)
rating_scaled_df.printSchema()


# rating_scaled_df.show(10)
# 把归一化结果转普通数值,而不是向量
@udf(returnType='float')
def unwrap_list(rating):
    return rating.toArray().tolist()[0]


rating_scaled_df = rating_scaled_df \
    .withColumn('ave_rating_minmax', unwrap_list(col('ave_rating_scaled')))
## rating_scaled_df.show(10)
# 去掉ave_rating/ave_rating_vec/ave_rating_scaled这些临时列
rating_result_df = rating_scaled_df \
    .select(['anime_id', 'ave_rating_minmax'])
result_df = anime_df \
    .join(rating_result_df, on='anime_id')  ## on,指定 键
result_df.printSchema()
## result_df.show(10)
# 保存数据
result_df \
    .write.format('csv') \
    .option('header',True) \
    .save('')

import { computed, defineComponent, inject, ref } from "vue";
import "./editor.scss";
import EditorBlock from "./editor-block";
import deepcopy from "deepcopy";
import { useMenuDrager } from "./useMenuDragger";
import { useFocus } from "./useFocus";
import { useBlockDragger } from "./useBlockDragger";
export default defineComponent({
  props: {
    // 接收父级传来的数据
    // v-model是语法糖,其传进来的props名称就是modelValue
    // <input modelValue="xxx" @change="xxx" />
    modelValue: { type: Object },
  },
  emits: ["update:modelValue"], // 要触发的事件
  setup(props, ctx) {
    const data = computed({
      get() {
        return props.modelValue;
      },
      set(newValue) {
        console.log("set");
        // console.log(newValue);
        // 触发事件更新数据,(v-model语法糖提供的)
        ctx.emit("update:modelValue", deepcopy(newValue));
      },
    });
    // console.log(props.data);
    // console.log(props.modelValue);
    // console.log(data.value);
    const containerStyles = computed(() => ({
      width: data.value.container.width + "px",
      height: data.value.container.height + "px",
    }));
    const config = inject("config");
    const containerRef = ref(null); // 获取内容区域对应的dom元素
    // 1,实现菜单拖拽功能
    const { dragstart, dragend } = useMenuDrager(containerRef, data);
    // 2,实现获取焦点
    // 实现组件拖拽
    let { blockMousedown, focusData } = useFocus(data, (e) => {
      // 有可能选中后直接进行拖拽
      // console.log(focusData.value.focus);
      mousedown(e);
    });
    // 3,实现拖拽多个元素的功能
    let { mousedown } = useBlockDragger(focusData);

    // 收集获取和没有获取焦点的组件(block)

    return () => (
      <div class="editor">
        <div class="editor-left">
          {/* 根据注册列表,渲染物料区,可以实现h5的拖拽 */}
          {config.componentList.map((component) => (

              draggable
              class="editor-left-item"
              onDrag={(e) => dragstart(e, component)}
              // onDragEnd={dragend} 原生的无法绑定,不知道为啥
            >
              <span>{component.label}</span>
              <div>{component.preview()}</div>
            </div>
          ))}

        <div class="editor-right">属性控制栏</div>

useBlockDragger.js

export function useBlockDragger(focusData) {
    // 记录起始位置
    let dragState = {
        startX: 0,
        startY: 0,
    };
    const mouseup = (e) => {
        document.removeEventListener("mousemove", mousemove);
        document.removeEventListener("mouseup", mouseup);
    };
    const mousemove = (e) => {
        // 获取移动后的距离
        let { clientX: moveX, clientY: moveY } = e;
        let durX = moveX - dragState.startX;
        let durY = moveY - dragState.startY;
        focusData.value.focus.map((block, idx) => {
            block.top = dragState.startPos[idx].top + durY;
            block.left = dragState.startPos[idx].left + durX;
        });
    };
    const mousedown = (e) => {
        dragState = {
            startX: e.clientX,
            startY: e.clientY,
            // 记录每一个选中的位置
            startPos: focusData.value.focus.map(({ top, left }) => ({ top, left })),
        };
        document.addEventListener("mousemove", mousemove);
        document.addEventListener("mouseup", mouseup);
    };
    return {
        mousedown
    }
}

useFocus.js

import { computed } from "vue";
export function useFocus(data, callback) {
  // 获取哪些元素被选中
  const focusData = computed(() => {
    let focus = [];
    let unfocused = [];
    data.value.blocks.forEach((block) =>
      (block.focus ? focus : unfocused).push(block)
    );
    return {
      focus,
      unfocused,
    };
  });
  const clearBlockFocus = () => {
    data.value.blocks.forEach((block) => (block.focus = false));
  };
  // Mousedown事件仅需要按键被按下，而不需要松开即可发生
  const blockMousedown = (e, block) => {
    e.preventDefault();
    e.stopPropagation();
    // block上我们定义一个属性focus,获取焦点后就将focus变为true
    // 如果是按住shift的情况,就可以多选
    if (e.shiftKey) {
      block.focus = !block.focus;
      callback(e);
      return;
    }
    if (!block.focus) {
      clearBlockFocus();
      block.focus = true; // 要清空其他block的focus,实现单选
      callback(e);
      return;
    }
    block.focus = false;
    callback(e);
  };
  const containerMouseDown = () => {
    clearBlockFocus(); // 点击容器,让选中的block失去焦点
  };
  // 不知道为啥直接onxxxx绑定失效了
  Promise.resolve().then(() => {
    document
      .querySelector(".editor-container-canvas__content")
      .addEventListener("mousedown", containerMouseDown);
  });
  return {
    blockMousedown,
    focusData,
  };
}

useMenuDragger.js

export function useMenuDrager(containerRef, data) {
  let currentComponent = null; // 记录拖拽的组件,drop时渲染
  const dragenter = (e) => {
    console.log("enter");
    // 该属性控制在拖放操作中给用户的反馈（通常是视觉上的）。它会影响在拖拽过程中光标的手势
    e.dataTransfer.dropEffect = "move";
  };
  const dragover = (e) => {
    e.preventDefault(); // !!!
    console.log("over");
  };
  const dragleave = (e) => {
    console.log("leave");
    e.dataTransfer.dropEffect = "none";
  };
  const drop = (e) => {
    console.log("drop");
    let blocks = data.value.blocks; // 内部已经渲染的组件
    // console.log(data.value);
    // 更新组件的属性
    data.value = {
      ...data.value,
      blocks: [
        ...blocks,
        {
          top: e.offsetY,
          left: e.offsetX,
          zIndex: 1,
          key: currentComponent.key,
          alignCenter: true, // 松手时居中
        },
      ],
    };
    currentComponent = null;
  };
  const dragstart = (e, component) => {
    // dragenter进入元素,添加移动标识
    // dragover经过元素,要阻止默认行为,负责不能触发drop
    // dragleave离开元素,增加禁用标识
    // drop松手时,根据拖拽的组件添加到内容区
    // console.log(containerRef.value);
    // e.dataTransfer.dropEffect = "move";
    containerRef.value.addEventListener("dragover", dragover);
    containerRef.value.addEventListener("dragenter", dragenter);
    containerRef.value.addEventListener("dragleave", dragleave);
    containerRef.value.addEventListener("drop", drop);
    currentComponent = component;
  };
  setTimeout(() => {
    document
      .querySelector(".editor-left-item")
      .addEventListener("dragend", dragend);
  }, 0);
  // 拖拽结束,清理事件
  const dragend = (e) => {
    console.log("end! remove listener");
    containerRef.value.removeEventListener("dragover", dragover);
    containerRef.value.removeEventListener("dragenter", dragenter);
    containerRef.value.removeEventListener("dragleave", dragleave);
    containerRef.value.removeEventListener("drop", drop);
  };
  return {
    dragstart,
    dragend,
  };
}

useBlockDragger.js

import { reactive } from "vue"; 
export function useBlockDragger(focusData, lastSelectBlock, data) {
    // 记录起始位置
    let dragState = {
        startX: 0,
        startY: 0,
    };
    // 响应式,让x,y双向绑定
    let markline = reactive({
        x: null,
        y: null
    })
    const mouseup = (e) => {
        document.removeEventListener("mousemove", mousemove);
        document.removeEventListener("mouseup", mouseup);
        markline.x = null
        markline.y = null
    };
    const mousemove = (e) => {
        // 获取移动后的距离
        let { clientX: moveX, clientY: moveY } = e;
        // 计算最新的left和top,去lines里找要显示的线
        // 鼠标移动后 - 鼠标移动前 + left
        let left = moveX - dragState.startX + dragState.startLeft
        let top = moveY - dragState.startY + dragState.startTop
        // 计算线 距离参照物还有5px就显示
        let x, y = null
        for (let i = 0; i < dragState.lines.y.length; i++) {
            const { top: t, showTop: s } = dragState.lines.y[i]
            if (Math.abs(t - top) < 5) {
                y = s // 线要显示的位置
                // 实现快速贴合
                // 容器距离顶部的距离+目标高度
                moveY = dragState.startY - dragState.startTop + t
                break // 找到一根线就跳出循环   
            }
        }
        for (let i = 0; i < dragState.lines.x.length; i++) {
            const { left: l, showLeft: s } = dragState.lines.x[i]
            if (Math.abs(l - left) < 5) {
                x = s // 线要显示的位置
                // 实现快速贴合
                // 容器距离顶部的距离+目标高度
                moveX = dragState.startX - dragState.startLeft + l
                break // 找到一根线就跳出循环   
            }
        }
        // 实现视图更新
        markline.x = x
        markline.y = y
        // .....
    };
    const mousedown = (e) => {
        // 获取宽高
        const { width: BWidth, height: BHeight } = lastSelectBlock.value // 拖拽的最后一个元素
        dragState = {
            startX: e.clientX,
            startY: e.clientY,
            startTop: lastSelectBlock.value.top, // 拖拽前的位置    
            startLeft: lastSelectBlock.value.left,
            // 记录每一个选中的位置
            startPos: focusData.value.focus.map(({ top, left }) => ({ top, left })),
            lines: (() => {
                const { unfocused } = focusData.value // 获取其他没选中的,以它们的位置做辅助
                let lines = { x: [], y: [] } // 计算横线的位置用y存放 x存纵向
                unfocused.push({
                    top: 0,
                    left: 0,
                    width: data.value.container.width,
                    height: data.value.container.height
                })
                // console.log(unfocused)
                unfocused.forEach((block) => {
                    const { top: ATop, left: ALeft, width: AWidth, height: AHeight } = block
                    // 当此元素拖动到A元素和top一致时,要显示这辅助线,辅助线的位置就是ATop
                    lines.y.push({ showTop: ATop, top: ATop }) // 顶对顶
                    lines.y.push({ showTop: ATop, top: ATop - BHeight }) // 顶对底
                    lines.y.push({ showTop: ATop + AHeight / 2, top: ATop - BHeight }) // 中对中
                    lines.y.push({ showTop: ATop + AHeight, top: ATop + AHeight }) // 底对顶
                    lines.y.push({ showTop: ATop + AHeight, top: ATop + AHeight + BHeight }) // 底对底
                    lines.x.push({ showLeft: ALeft, left: ALeft }) // 左对左
                    lines.x.push({ showLeft: ALeft + AWidth, left: ALeft + AWidth }) // 右对左
                    lines.x.push({ showLeft: ALeft + AWidth / 2, left: ALeft + AWidth / 2 - BWidth / 2 }) // 中对中
                    lines.x.push({ showLeft: ALeft + AWidth, left: ALeft + AWidth - BWidth }) // 右对右
                    lines.x.push({ showLeft: ALeft, left: ALeft + AWidth - BWidth }) // 左对右
                })
                // console.log(lines) 
                return lines
            })()
        }
        document.addEventListener("mousemove", mousemove);
        document.addEventListener("mouseup", mouseup);
    }
    return {
        mousedown,
        markline
    }
}

useFocus.js 获取最后一个拖拽元素

import { computed, ref } from "vue";
export function useFocus(data, callback) {
  const selectIndex = ref(-1) // 记录最后点击的组件的索引
  const lastSelectBlock = computed(() => data.value.blocks[selectIndex.value]) // 最后点击的元素
  // 获取哪些元素被选中
  const focusData = computed(() => {
    let focus = [];
    let unfocused = [];
    data.value.blocks.forEach((block) =>
      (block.focus ? focus : unfocused).push(block)
    );
    return {
      focus,
      unfocused,
    };
  });
  const clearBlockFocus = () => {
    data.value.blocks.forEach((block) => (block.focus = false));
    selectIndex.value = -1
  };
  // Mousedown事件仅需要按键被按下，而不需要松开即可发生
  const blockMousedown = (e, block, index) => {
    e.preventDefault();
    e.stopPropagation();
    // block上我们定义一个属性focus,获取焦点后就将focus变为true
    // 如果是按住shift的情况,就可以多选
    if (e.shiftKey) {
      // 如果只有一个选中的block,每次点击时就不需要切换focus状态
      if (focusData.value.focus.length <= 1) {
        block.focus = true
        selectIndex.value = index
        callback(e);
        return
      }
      block.focus = !block.focus;
      selectIndex.value = index
      callback(e);
      return;
    }
    if (!block.focus) {
      clearBlockFocus();
      block.focus = true; // 要清空其他block的focus,实现单选
      selectIndex.value = index
      callback(e);
      return;
    }
    // block.focus = false; // 已经被选中,则再次点击时还是选中状态
    selectIndex.value = index
    callback(e);
  };
  // .....
  return {
    blockMousedown,
    focusData,
    lastSelectBlock
  };
}

      },
      set(newValue) {
        console.log("set");
        // console.log(newValue);
        // 触发事件更新数据,(v-model语法糖提供的)
        ctx.emit("update:modelValue", deepcopy(newValue));
      },
    });
    // console.log(props.data);
    // console.log(props.modelValue);
    // console.log(data.value);
    const containerStyles = computed(() => ({
      width: data.value.container.width + "px",
      height: data.value.container.height + "px",
    }));
    const config = inject("config");
    const containerRef = ref(null); // 获取内容区域对应的dom元素
    // 1,实现菜单拖拽功能
    const { dragstart, dragend } = useMenuDrager(containerRef, data);
    // 2,实现获取焦点
    // 实现组件拖拽
    let { blockMousedown, focusData, lastSelectBlock } = useFocus(data, (e) => {
      // 有可能选中后直接进行拖拽
      // console.log(focusData.value.focus);
      mousedown(e);
    });
    // 3,实现拖拽多个元素的功能
    let { mousedown, markline } = useBlockDragger(
      focusData,
      lastSelectBlock,
      data
    );

    // 收集获取和没有获取焦点的组件(block)

    return () => (
      <div class="editor">
        <div class="editor-left">
          {/* 根据注册列表,渲染物料区,可以实现h5的拖拽 */}
          {config.componentList.map((component) => (
            <div
              draggable
              class="editor-left-item"
              onDrag={(e) => dragstart(e, component)}
              // onDragEnd={dragend} 原生的无法绑定,不知道为啥
            >
              <span>{component.label}</span>
              <div>{component.preview()}</div>
            </div>
          ))}
        </div>
        <div class="editor-top">菜单栏</div>
        <div class="editor-right">属性控制栏</div>
        <div class="editor-container">
          {/* 负责产生滚动条 */}
          <div class="editor-container-canvas">
            {/* 产生内容区域 */}
            <div
              class="editor-container-canvas__content"
              ref={containerRef}
              style={containerStyles.value}
              // 点击组件以外的内容区,可以把focus取消
              // onMouseDown={containerMouseDown}
            >
              {data.value.blocks.map((block, index) => (
                <EditorBlock
                  class={block.focus ? "editor-block-focus" : ""} // focus的话需要添加样式
                  block={block}
                  onMousedown={(e) => blockMousedown(e, block, index)}
                ></EditorBlock>
              ))}
              {markline.x != null && (
                <div class="line-x" style={{ left: markline.x + "px" }}></div>
              )}
              {markline.y != null && (
                <div class="line-y" style={{ top: markline.y + "px" }}></div>
              )}
            </div>
          </div>
        </div>
      </div>
    );
  },
});
